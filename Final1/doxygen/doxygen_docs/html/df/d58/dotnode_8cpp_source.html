100 static QCString escapeTooltip(const QCString &tooltip)
101 {
102  QCString result;
103  const char *p=tooltip.data();
104  if (p==0) return result;
105  char c;
106  while ((c=*p++))
107  {
108  switch(c)
109  {
110  case '"': result+="\\\""; break;
111  case '\\': result+="\\\\"; break;
112  default: result+=c; break;
113  }
114  }
115  return result;
116 }

118 static void writeBoxMemberList(FTextStream &t,
119  char prot,MemberList *ml,const ClassDef *scope,
120  bool isStatic=FALSE,const StringUnorderedSet *skipNames=nullptr)
121 {
122  if (ml)
123  {
124  MemberListIterator mlia(*ml);
125  MemberDef *mma;
126  int totalCount=0;
127  for (mlia.toFirst();(mma = mlia.current());++mlia)
128  {
129  if (mma->getClassDef()==scope &&
130  (skipNames==nullptr || skipNames->find(mma->name().str())==std::end(*skipNames)))
131  {
132  totalCount++;
133  }
134  }
135 
136  int count=0;
137  for (mlia.toFirst();(mma = mlia.current());++mlia)
138  {
139  if (mma->getClassDef() == scope &&
140  (skipNames==nullptr || skipNames->find(mma->name().str())==std::end(*skipNames)))
141  {
142  int numFields = Config_getInt(UML_LIMIT_NUM_FIELDS);
143  if (numFields>0 && (totalCount>numFields*3/2 && count>=numFields))
144  {
145  t << theTranslator->trAndMore(QCString().sprintf("%d",totalCount-count)) << "\\l";
146  break;
147  }
148  else
149  {
150  t << prot << " ";
151  t << DotNode::convertLabel(mma->name());
152  if (!mma->isObjCMethod() &&
153  (mma->isFunction() || mma->isSlot() || mma->isSignal())) t << "()";
154  t << "\\l";
155  count++;
156  }
157  }
158  }
159  // write member groups within the memberlist
160  MemberGroupList *mgl = ml->getMemberGroupList();
161  if (mgl)
162  {
163  MemberGroupListIterator mgli(*mgl);
164  MemberGroup *mg;
165  for (mgli.toFirst();(mg=mgli.current());++mgli)
166  {
167  if (mg->members())
168  {
169  writeBoxMemberList(t,prot,mg->members(),scope,isStatic,skipNames);
170  }
171  }
172  }
173  }

176 QCString DotNode::convertLabel(const QCString &l)
177 {
178  QCString bBefore("\\_/<({[: =-+@%#~?$"); // break before character set
179  QCString bAfter(">]),:;|"); // break after character set
180  QCString p(l);
181  if (p.isEmpty()) return QCString();
182  QCString result;
183  char c,pc=0;
184  uint idx = 0;
185  int len=p.length();
186  int charsLeft=len;
187  int sinceLast=0;
188  int foldLen=17; // ideal text length
189  while (idx < p.length())
190  {
191  c = p[idx++];
192  QCString replacement;
193  switch(c)
194  {
195  case '\\': replacement="\\\\"; break;
196  case '\n': replacement="\\n"; break;
197  case '<': replacement="\\<"; break;
198  case '>': replacement="\\>"; break;
199  case '|': replacement="\\|"; break;
200  case '{': replacement="\\{"; break;
201  case '}': replacement="\\}"; break;
202  case '"': replacement="\\\""; break;
203  default: replacement+=c; break;
204  }
205  // Some heuristics to insert newlines to prevent too long
206  // boxes and at the same time prevent ugly breaks
207  if (c=='\n')
208  {
209  result+=replacement;
210  foldLen = (3*foldLen+sinceLast+2)/4;
211  sinceLast=1;
212  }
213  else if ((pc!=':' || c!=':') && charsLeft>foldLen/3 && sinceLast>foldLen && bBefore.contains(c))
214  {
215  result+="\\l";
216  result+=replacement;
217  foldLen = (foldLen+sinceLast+1)/2;
218  sinceLast=1;
219  }
220  else if (charsLeft>1+foldLen/4 && sinceLast>foldLen+foldLen/3 &&
221  !isupper(c) && isupper(p[idx]))
222  {
223  result+=replacement;
224  result+="\\l";
225  foldLen = (foldLen+sinceLast+1)/2;
226  sinceLast=0;
227  }
228  else if (charsLeft>foldLen/3 && sinceLast>foldLen && bAfter.contains(c) && (c!=':' || p[idx]!=':'))
229  {
230  result+=replacement;
231  result+="\\l";
232  foldLen = (foldLen+sinceLast+1)/2;
233  sinceLast=0;
234  }
235  else
236  {
237  result+=replacement;
238  sinceLast++;
239  }
240  charsLeft--;
241  pc=c;
242  }
243  return result;
244 }

246 static QCString stripProtectionPrefix(const QCString &s)
247 {
248  if (!s.isEmpty() && (s[0]=='-' || s[0]=='+' || s[0]=='~' || s[0]=='#'))
249  {
250  return s.mid(1);
251  }
252  else
253  {
254  return s;
255  }

319 void DotNode::deleteNode(DotNodeList &deletedList,SDict<DotNode> *skipNodes)
320 {
321  if (m_deleted) return; // avoid recursive loops in case the graph has cycles
322  m_deleted=TRUE;
323  if (m_parents!=0) // delete all parent nodes of this node
324  {
325  QListIterator<DotNode> dnlip(*m_parents);
326  DotNode *pn;
327  for (dnlip.toFirst();(pn=dnlip.current());++dnlip)
328  {
329  //pn->removeChild(this);
330  pn->deleteNode(deletedList,skipNodes);
331  }
332  }
333  if (m_children!=0) // delete all child nodes of this node
334  {
335  QListIterator<DotNode> dnlic(*m_children);
336  DotNode *cn;
337  for (dnlic.toFirst();(cn=dnlic.current());++dnlic)
338  {
339  //cn->removeParent(this);
340  cn->deleteNode(deletedList,skipNodes);
341  }
342  }
343  // add this node to the list of deleted nodes.
344  //printf("skipNodes=%p find(%p)=%p\n",skipNodes,this,skipNodes ? skipNodes->find((int)this) : 0);
345  if (skipNodes==0 || skipNodes->find((char*)this)==0)
346  {
347  //printf("deleting\n");
348  deletedList.append(this);
349  }
350 }

375 void DotNode::writeBox(FTextStream &t,
376  GraphType gt,
377  GraphOutputFormat /*format*/,
378  bool hasNonReachableChildren) const
379 {
380  const char *labCol =
381  m_url.isEmpty() ? "grey75" : // non link
382  (hasNonReachableChildren ? "red" : "black");
383  t << " Node" << m_number << " [label=\"";
384 
385  if (m_classDef && Config_getBool(UML_LOOK) && (gt==Inheritance || gt==Collaboration))
386  {
387  // add names shown as relations to a set, so we don't show
388  // them as attributes as well
389  StringUnorderedSet arrowNames;
390  if (m_edgeInfo)
391  {
392  // for each edge
393  QListIterator<EdgeInfo> li(*m_edgeInfo);
394  EdgeInfo *ei;
395  for (li.toFirst();(ei=li.current());++li)
396  {
397  if (!ei->label().isEmpty()) // labels joined by \n
398  {
399  int i=ei->label().find('\n');
400  int p=0;
401  QCString lab;
402  while ((i=ei->label().find('\n',p))!=-1)
403  {
404  lab = stripProtectionPrefix(ei->label().mid(p,i-p));
405  arrowNames.insert(lab.str());
406  p=i+1;
407  }
408  lab = stripProtectionPrefix(ei->label().right(ei->label().length()-p));
409  arrowNames.insert(lab.str());
410  }
411  }
412  }
413 
414  //printf("DotNode::writeBox for %s\n",m_classDef->name().data());
415  t << "{" << convertLabel(m_label);
416  t << "\\n|";
417  writeBoxMemberList(t,'+',m_classDef->getMemberList(MemberListType_pubAttribs),m_classDef,FALSE,&arrowNames);
418  writeBoxMemberList(t,'+',m_classDef->getMemberList(MemberListType_pubStaticAttribs),m_classDef,TRUE,&arrowNames);
419  writeBoxMemberList(t,'+',m_classDef->getMemberList(MemberListType_properties),m_classDef,FALSE,&arrowNames);
420  writeBoxMemberList(t,'~',m_classDef->getMemberList(MemberListType_pacAttribs),m_classDef,FALSE,&arrowNames);
421  writeBoxMemberList(t,'~',m_classDef->getMemberList(MemberListType_pacStaticAttribs),m_classDef,TRUE,&arrowNames);
422  writeBoxMemberList(t,'#',m_classDef->getMemberList(MemberListType_proAttribs),m_classDef,FALSE,&arrowNames);
423  writeBoxMemberList(t,'#',m_classDef->getMemberList(MemberListType_proStaticAttribs),m_classDef,TRUE,&arrowNames);
424  if (Config_getBool(EXTRACT_PRIVATE))
425  {
426  writeBoxMemberList(t,'-',m_classDef->getMemberList(MemberListType_priAttribs),m_classDef,FALSE,&arrowNames);
427  writeBoxMemberList(t,'-',m_classDef->getMemberList(MemberListType_priStaticAttribs),m_classDef,TRUE,&arrowNames);
428  }
429  t << "|";
430  writeBoxMemberList(t,'+',m_classDef->getMemberList(MemberListType_pubMethods),m_classDef);
431  writeBoxMemberList(t,'+',m_classDef->getMemberList(MemberListType_pubStaticMethods),m_classDef,TRUE);
432  writeBoxMemberList(t,'+',m_classDef->getMemberList(MemberListType_pubSlots),m_classDef);
433  writeBoxMemberList(t,'~',m_classDef->getMemberList(MemberListType_pacMethods),m_classDef);
434  writeBoxMemberList(t,'~',m_classDef->getMemberList(MemberListType_pacStaticMethods),m_classDef,TRUE);
435  writeBoxMemberList(t,'#',m_classDef->getMemberList(MemberListType_proMethods),m_classDef);
436  writeBoxMemberList(t,'#',m_classDef->getMemberList(MemberListType_proStaticMethods),m_classDef,TRUE);
437  writeBoxMemberList(t,'#',m_classDef->getMemberList(MemberListType_proSlots),m_classDef);
438  if (Config_getBool(EXTRACT_PRIVATE))
439  {
440  writeBoxMemberList(t,'-',m_classDef->getMemberList(MemberListType_priMethods),m_classDef);
441  writeBoxMemberList(t,'-',m_classDef->getMemberList(MemberListType_priStaticMethods),m_classDef,TRUE);
442  writeBoxMemberList(t,'-',m_classDef->getMemberList(MemberListType_priSlots),m_classDef);
443  }
444  if (m_classDef->getLanguage()!=SrcLangExt_Fortran &&
445  m_classDef->getMemberGroupSDict())
446  {
447  MemberGroupSDict::Iterator mgdi(*m_classDef->getMemberGroupSDict());
448  MemberGroup *mg;
449  for (mgdi.toFirst();(mg=mgdi.current());++mgdi)
450  {
451  if (mg->members())
452  {
453  writeBoxMemberList(t,'*',mg->members(),m_classDef,FALSE,&arrowNames);
454  }
455  }
456  }
457  t << "}";
458  }
459  else // standard look
460  {
461  t << convertLabel(m_label);
462  }
463  t << "\",height=0.2,width=0.4";
464  if (m_isRoot)
465  {
466  t << ",color=\"black\", fillcolor=\"grey75\", style=\"filled\", fontcolor=\"black\"";
467  }
468  else
469  {
470  if (!Config_getBool(DOT_TRANSPARENT))
471  {
472  t << ",color=\"" << labCol << "\", fillcolor=\"";
473  t << "white";
474  t << "\", style=\"filled\"";
475  }
476  else
477  {
478  t << ",color=\"" << labCol << "\"";
479  }
480  if (!m_url.isEmpty())
481  {
482  int anchorPos = m_url.findRev('#');
483  if (anchorPos==-1)
484  {
485  t << ",URL=\"" << m_url << Doxygen::htmlFileExtension << "\"";
486  }
487  else
488  {
489  t << ",URL=\"" << m_url.left(anchorPos) << Doxygen::htmlFileExtension
490  << m_url.right(m_url.length()-anchorPos) << "\"";
491  }
492  }
493  }
494  if (!m_tooltip.isEmpty())
495  {
496  t << ",tooltip=\"" << escapeTooltip(m_tooltip) << "\"";
497  }
498  else
499  {
500  t << ",tooltip=\" \""; // space in tooltip is required otherwise still something like 'Node0' is used
501  }
502  t << "];" << endl;
503 }

505 void DotNode::writeArrow(FTextStream &t,
506  GraphType gt,
507  GraphOutputFormat format,
508  const DotNode *cn,
509  const EdgeInfo *ei,
510  bool topDown,
511  bool pointBack) const

513  t << " Node";
514  if (topDown)
515  t << cn->number();
516  else
517  t << m_number;
518  t << " -> Node";
519  if (topDown)
520  t << m_number;
521  else
522  t << cn->number();
523  t << " [";
524 
525  const EdgeProperties *eProps = Config_getBool(UML_LOOK) ? &umlEdgeProps : &normalEdgeProps;
526  QCString aStyle = eProps->arrowStyleMap[ei->color()];
527  bool umlUseArrow = aStyle=="odiamond";
528 
529  if (pointBack && !umlUseArrow) t << "dir=\"back\",";
530  t << "color=\"" << eProps->edgeColorMap[ei->color()]
531  << "\",fontsize=\"" << Config_getInt(DOT_FONTSIZE) << "\",";
532  t << "style=\"" << eProps->edgeStyleMap[ei->style()] << "\"";
533  if (!ei->label().isEmpty())
534  {
535  t << ",label=\" " << convertLabel(ei->label()) << "\" ";
536  }
537  if (Config_getBool(UML_LOOK) &&
538  eProps->arrowStyleMap[ei->color()] &&
539  (gt==Inheritance || gt==Collaboration)
540  )
541  {
542  bool rev = pointBack;
543  if (umlUseArrow) rev=!rev; // UML use relates has arrow on the start side
544  if (rev)
545  t << ",arrowtail=\"" << eProps->arrowStyleMap[ei->color()] << "\"";
546  else
547  t << ",arrowhead=\"" << eProps->arrowStyleMap[ei->color()] << "\"";
548  }
549 
550  if (format==GOF_BITMAP) t << ",fontname=\"" << Config_getString(DOT_FONTNAME) << "\"";
551  t << "];" << endl;

554 void DotNode::write(FTextStream &t,
555  GraphType gt,
556  GraphOutputFormat format,
557  bool topDown,
558  bool toChildren,
559  bool backArrows) const
560 {
561  //printf("DotNode::write(%d) name=%s this=%p written=%d visible=%d\n",m_distance,m_label.data(),this,m_written,m_visible);
562  if (m_written) return; // node already written to the output
563  if (!m_visible) return; // node is not visible
564  writeBox(t,gt,format,m_truncated==Truncated);
565  m_written=TRUE;
566  QList<DotNode> *nl = toChildren ? m_children : m_parents;
567  if (nl)
568  {
569  if (toChildren)
570  {
571  QListIterator<DotNode> dnli1(*nl);
572  QListIterator<EdgeInfo> dnli2(*m_edgeInfo);
573  const DotNode *cn;
574  for (dnli1.toFirst();(cn=dnli1.current());++dnli1,++dnli2)
575  {
576  if (cn->isVisible())
577  {
578  //printf("write arrow %s%s%s\n",label().data(),backArrows?"<-":"->",cn->label().data());
579  writeArrow(t,gt,format,cn,dnli2.current(),topDown,backArrows);
580  }
581  cn->write(t,gt,format,topDown,toChildren,backArrows);
582  }
583  }
584  else // render parents
585  {
586  QListIterator<DotNode> dnli(*nl);
587  DotNode *pn;
588  for (dnli.toFirst();(pn=dnli.current());++dnli)
589  {
590  if (pn->isVisible())
591  {
592  //printf("write arrow %s%s%s\n",label().data(),backArrows?"<-":"->",pn->label().data());
593  writeArrow(t,
594  gt,
595  format,
596  pn,
597  pn->edgeInfo()->at(pn->children()->findRef(this)),
598  FALSE,
599  backArrows
600  );
601  }
602  pn->write(t,gt,format,TRUE,FALSE,backArrows);
603  }
604  }
605  }
606  //printf("end DotNode::write(%d) name=%s\n",distance,m_label.data());
607 }

609 void DotNode::writeXML(FTextStream &t,bool isClassGraph) const
610 {
611  t << " <node id=\"" << m_number << "\">" << endl;
612  t << " <label>" << convertToXML(m_label) << "</label>" << endl;
613  if (!m_url.isEmpty())
614  {
615  QCString url(m_url);
616  const char *refPtr = url.data();
617  char *urlPtr = strchr(url.rawData(),'$');
618  if (urlPtr)
619  {
620  *urlPtr++='\0';
621  t << " <link refid=\"" << convertToXML(urlPtr) << "\"";
622  if (*refPtr!='\0')
623  {
624  t << " external=\"" << convertToXML(refPtr) << "\"";
625  }
626  t << "/>" << endl;
627  }
628  }
629  if (m_children)
630  {
631  QListIterator<DotNode> nli(*m_children);
632  QListIterator<EdgeInfo> eli(*m_edgeInfo);
633  DotNode *childNode;
634  EdgeInfo *edgeInfo;
635  for (;(childNode=nli.current());++nli,++eli)
636  {
637  edgeInfo=eli.current();
638  t << " <childnode refid=\"" << childNode->number() << "\" relation=\"";
639  if (isClassGraph)
640  {
641  switch(edgeInfo->color())
642  {
643  case EdgeInfo::Blue: t << "public-inheritance"; break;
644  case EdgeInfo::Green: t << "protected-inheritance"; break;
645  case EdgeInfo::Red: t << "private-inheritance"; break;
646  case EdgeInfo::Purple: t << "usage"; break;
647  case EdgeInfo::Orange: t << "template-instance"; break;
648  case EdgeInfo::Orange2: t << "type-constraint"; break;
649  case EdgeInfo::Grey: ASSERT(0); break;
650  }
651  }
652  else // include graph
653  {
654  t << "include";
655  }
656  t << "\">" << endl;
657  if (!edgeInfo->label().isEmpty())
658  {
659  int p=0;
660  int ni;
661  while ((ni=edgeInfo->label().find('\n',p))!=-1)
662  {
663  t << " <edgelabel>"
664  << convertToXML(edgeInfo->label().mid(p,ni-p))
665  << "</edgelabel>" << endl;
666  p=ni+1;
667  }
668  t << " <edgelabel>"
669  << convertToXML(edgeInfo->label().right(edgeInfo->label().length()-p))
670  << "</edgelabel>" << endl;
671  }
672  t << " </childnode>" << endl;
673  }
674  }
675  t << " </node>" << endl;
